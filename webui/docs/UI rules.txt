Method 1: The "Svelte Logic" Way (Easiest to read)
Svelte has a special tag called <svelte:window> that lets you track the browser size in real-time. You can use this to literally swap the components based on the pixel width.

The Logic:

If the screen is wider than 768px (standard tablet/desktop cutoff), show <Sidebar />.

Else, show <BottomNav />.

The Code:
<script>
  import Sidebar from './Sidebar.svelte';
  import BottomNav from './BottomNav.svelte';

  // This variable automatically updates whenever the window resizes
  let innerWidth;
</script>

<svelte:window bind:innerWidth />

<main class="layout">
  {#if innerWidth >= 768}
    <div class="desktop-layout">
      <Sidebar />
      <div class="content">
        <slot /> </div>
    </div>
  {:else}
    <div class="mobile-layout">
      <div class="content">
        <slot />
      </div>
      <BottomNav />
    </div>
  {/if}
</main>

import requests
import time

def get_current_user_playlists_v2(access_token):
    base_url = "https://openapi.tidal.com/v2"
    headers = {
        "Authorization": f"Bearer {access_token}",
        "Accept": "application/vnd.api+json",
        "Content-Type": "application/vnd.api+json"
    }

    # --- STEP 1: Get User ID ---
    print("Fetching User ID...")
    me_resp = requests.get(f"{base_url}/users/me", headers=headers)
    
    if me_resp.status_code != 200:
        raise Exception(f"Failed to fetch user: {me_resp.text}")
    
    # Parse JSON:API response for ID
    me_data = me_resp.json().get('data', {})
    user_id = me_data.get('id')
    
    # Try to grab country code from attributes, default to US if missing
    attributes = me_data.get('attributes', {})
    country_code = attributes.get('countryCode', 'US')
    
    print(f"Found User ID: {user_id} (Country: {country_code})")

    # --- STEP 2: Get Playlists ---
    # Note: We use the 'relationships/playlists' endpoint
    playlists_url = f"{base_url}/users/{user_id}/relationships/playlists"
    params = {
        "countryCode": country_code,
        "page[limit]": 50  # JSON:API pagination limit
    }

    print(f"Fetching playlists from: {playlists_url}")
    resp = requests.get(playlists_url, headers=headers, params=params)

    if resp.status_code != 200:
        raise Exception(f"Failed to fetch playlists: {resp.text}")

    # Parse JSON:API response
    # The 'data' array contains the playlist objects
    playlist_data = resp.json().get('data', [])
    
    results = []
    for item in playlist_data:
        attr = item.get('attributes', {})
        results.append({
            "id": item.get('id'),
            "name": attr.get('title', 'Unknown Playlist'),
            "description": attr.get('description'),
            "last_updated": attr.get('lastModifiedAt')
        })
        
    return results
